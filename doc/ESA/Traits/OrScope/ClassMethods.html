def or_scope(*scopes)
  conditions = 
    scopes
      .map { |scope| "(#{scope.where_clauses.map{ |clause| "(#{clause})"}.join(" AND ")})" }
      .join(" OR ")
      
  relationships =
    scopes
      .map { |scope| scope.joins_values }
      .flatten
      .uniq
      
  joins(*relationships).where(conditions)
end
